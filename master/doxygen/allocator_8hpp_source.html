244  pointer
245  allocate(size_type cnt, const_void_pointer = 0)
246  {
247  if (pmemobj_tx_stage() != TX_STAGE_WORK)
248  throw pmem::transaction_scope_error(
249  "refusing to allocate memory outside of transaction scope");
250 
251  /* allocate raw memory, no object construction */
252  pointer ptr = pmemobj_tx_alloc(sizeof(value_type) * cnt,
253  detail::type_num<value_type>());
254 
255  if (ptr == nullptr) {
256  if (errno == ENOMEM) {
257  throw pmem::transaction_out_of_memory(
258  "Failed to allocate persistent memory object")
259  .with_pmemobj_errormsg();

261  throw pmem::transaction_alloc_error(

263  .with_pmemobj_errormsg();
264  }
