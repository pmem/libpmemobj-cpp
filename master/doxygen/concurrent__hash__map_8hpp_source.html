118  void
119  acquire(rw_mutex_type &m, bool write = true)
120  {
121  is_writer = write;
122  mutex = &m;
123  if (write)
124  mutex->lock();
125  else
126  mutex->lock_shared();
127  }
128 
132  void
133  release()
134  {
135  assert(mutex);
136  rw_mutex_type *m = mutex;
137  mutex = nullptr;
138  if (is_writer) {
139  m->unlock();
140  } else {
141  m->unlock_shared();
142  }

151  {
152  assert(!mutex);
153  bool result;
154  is_writer = write;
155  result = write ? m.try_lock() : m.try_lock_shared();
156  if (result)

653  {

655

666  void
667  disable()
668  {
669  assert(my_seg >= embedded_segments);
670 
671  if (my_seg < first_block) {
672  if (my_seg == embedded_segments) {
673  size_type sz = segment_size(first_block) -
674  embedded_buckets;
675  delete_persistent<bucket_type[]>(
676  (*my_table)[my_seg], sz);
677  }
678  (*my_table)[my_seg] = nullptr;
679  } else {
680  block_range blocks = segment_blocks(my_seg);
681 
682  for (segment_index_t b = blocks.first;
683  b < blocks.second; ++b) {
684  if ((*my_table)[b] != nullptr) {
685  delete_persistent<bucket_type[]>(
686  (*my_table)[b], block_size(b));
687  (*my_table)[b] = nullptr;
688  }
689  }
690  }
