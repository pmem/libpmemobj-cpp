806  * Need to descend the tree twice. First to find a leaf that
807  * represents a subtree that shares a common prefix with the key.
808  * This is needed to find out the actual labels between nodes (they
809  * are not known due to a possible path compression). Second time to
810  * find the place for the new element.
811  */
812  auto leaf = descend(key);
813  auto leaf_key = bytes_view(leaf->key());
814  auto diff = prefix_diff(key, leaf_key);
815 
816  auto n = root;
817  auto child_slot = &root;
818  auto prev = n;
819 
820  auto min_key_len = (std::min)(leaf_key.size(), key.size());
821  bitn_t sh = 8;
822 
823  /* If key differs from leaf_key at some point (neither is a prefix of
824  * another) we will descend to the point of divergence. Otherwise we
825  * will look for a node which represents the prefix. */
826  if (diff < min_key_len) {
827  auto at =
828  static_cast<unsigned char>(leaf_key[diff] ^ key[diff]);
829  sh = pmem::detail::mssb_index((uint32_t)at) & SLICE_MASK;
830  }
831 
832  /* Descend into the tree again. */
833  while (n && !n.is_leaf() &&
834  (n->byte < diff || (n->byte == diff && n->bit >= sh))) {
835  prev = n;
836  child_slot = &n->child[slice_index(key[n->byte], n->bit)];
837  n = *child_slot;
838  }
839 
840  /*
841  * If the divergence point is at same nib as an existing node, and
842  * the subtree there is empty, just place our leaf there and we're
843  * done. Obviously this can't happen if SLICE == 1.
844  */
845  if (!n) {
846  assert(diff < min_key_len);
847 
848  transaction::run(pop, [&] { *child_slot = make_leaf(prev); });
849  return {iterator(child_slot->get_leaf(), &root), true};
850  }
851 
852  /* New key is a prefix of the leaf key or they are equal. We need to add
853  * leaf ptr to internal node. */
854  if (diff == key.size()) {
855  // XXX we can just return leaf before the second loop.
856  if (n.is_leaf() &&
857  bytes_view(n.get_leaf()->key()).size() == key.size()) {
858  /* Key exists. */
859  return {iterator(child_slot->get_leaf(), &root), false};
860  }
861 
862  if (!n.is_leaf() && path_length_equal(key.size(), n)) {
863  if (n->embedded_entry)
864  return {iterator(n->embedded_entry.get_leaf(),
865  &root),
866  false};
867 
868  transaction::run(
869  pop, [&] { n->embedded_entry = make_leaf(n); });
870 
871  return {iterator(n->embedded_entry.get_leaf(), &root),
872  true};
873  }
874 
875  /* Path length from root to n is longer than key.size().
876  * We have to allocate new internal node above n. */
877  tagged_node_ptr node;
878  transaction::run(pop, [&] {
879  node = make_persistent<radix_tree::node>(
880  parent_ref(n), diff, bitn_t(FIRST_NIB));
881  node->embedded_entry = make_leaf(node);
882  node->child[slice_index(leaf_key[diff],
883  bitn_t(FIRST_NIB))] = n;
884 
885  parent_ref(n) = node;
886  *child_slot = node;
887  });
888 
889  return {iterator(node->embedded_entry.get_leaf(), &root), true};
890  }
891 
892  if (diff == leaf_key.size()) {
893  /* Leaf key is a prefix of the new key. We need to convert leaf
894  * to a node. */
895  tagged_node_ptr node;
896  transaction::run(pop, [&] {
897  /* We have to add new node at the edge from parent to n
898  */
899  node = make_persistent<radix_tree::node>(
900  parent_ref(n), diff, bitn_t(FIRST_NIB));
901  node->embedded_entry = n;
902  node->child[slice_index(key[diff], bitn_t(FIRST_NIB))] =
903  make_leaf(node);
904 
905  parent_ref(n) = node;
906  *child_slot = node;
907  });
908 
909  return {iterator(node->child[slice_index(key[diff],
910  bitn_t(FIRST_NIB))]
911  .get_leaf(),
912  &root),
913  true};
914  }
915 
916  /* There is already a subtree at the divergence point
917  * (slice_index(key[diff], sh)). This means that a tree is vertically
918  * compressed and we have to "break" this compression and add a new
919  * node. */
920  tagged_node_ptr node;
921  transaction::run(pop, [&] {
922  node = make_persistent<radix_tree::node>(parent_ref(n), diff,
923  sh);
924  node->child[slice_index(leaf_key[diff], sh)] = n;
925  node->child[slice_index(key[diff], sh)] = make_leaf(node);
926 
927  parent_ref(n) = node;
928  *child_slot = node;
929  });
930 
931  return {iterator(node->child[slice_index(key[diff], sh)].get_leaf(),
932  &root),
933  true};
934 }
