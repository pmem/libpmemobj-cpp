492 template <typename CharT, typename Traits>
493 typename basic_string_view<CharT, Traits>::size_type
494 basic_string_view<CharT, Traits>::find(const CharT *s, size_type pos,
495  size_type count) const
496 {
497  auto sz = size();
498 
499  if (pos > sz)
500  return npos;
501 
502  if (count == 0)
503  return pos;
504 
505  while (pos + count <= sz) {
506  auto found = traits_type::find(data() + pos, sz - pos, s[0]);
507  if (!found)
508  return npos;
509  pos = static_cast<size_type>(std::distance(data(), found));
510  if (traits_type::compare(found, s, count) == 0) {
511  return pos;
512  }
513  ++pos;
514  }
515  return npos;
516 }
