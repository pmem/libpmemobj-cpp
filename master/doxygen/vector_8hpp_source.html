654 template <typename T>
655 void
656 vector<T>::assign(size_type count, const_reference value)
657 {
658  pool_base pb = get_pool();
659 
660  transaction::run(pb, [&] {
661  if (count <= capacity()) {
662  /*
663  * Reallocation is not needed. First, replace old
664  * elements with new ones in range [0, size()).
665  * Depending on count, either call remaining old
666  * elements destructors, or append more new elements.
667  */
668  size_type size_old = _size;
669  add_data_to_tx(0, size_old);
670 
671  std::fill_n(
672  &_data[0],
673  (std::min)(count,
674  static_cast<size_type>(size_old)),
675  value);
676 
677  if (count > size_old) {
678  add_data_to_tx(size_old, count - size_old);
679  construct_at_end(count - size_old, value);
680  } else {
681  shrink(count);
682  }
683  } else {
684  dealloc();
685  alloc(count);
686  construct_at_end(count, value);
687  }
688  });
689 }

709 template <typename T>
710 template <typename InputIt,
711  typename std::enable_if<detail::is_input_iterator<InputIt>::value,
712  InputIt>::type *>
713 void
714 vector<T>::assign(InputIt first, InputIt last)
715 {
716  pool_base pb = get_pool();
717 
718  size_type size_new = static_cast<size_type>(std::distance(first, last));
719 
720  transaction::run(pb, [&] {
721  if (size_new <= capacity()) {
722  /*
723  * Reallocation is not needed. First, replace old
724  * elements with new ones in range [0, size()).
725  * Depending on size_new, either call remaining old
726  * elements destructors, or append more new elements.
727  */
728  size_type size_old = _size;
729  add_data_to_tx(0, size_old);
730 
731  InputIt mid = last;
732  bool growing = size_new > size_old;
733 
734  if (growing) {
735  add_data_to_tx(size_old, size_new - size_old);
736 
737  mid = first;
738  std::advance(mid, size_old);
739  }
740 
741  iterator shrink_to = std::copy(first, mid, &_data[0]);
742 
743  if (growing) {
744  construct_at_end(mid, last);
745  } else {
746  shrink(static_cast<size_type>(std::distance(
747  iterator(&_data[0]), shrink_to)));
748  }
749  } else {
750  dealloc();
751  alloc(size_new);
752  construct_at_end(first, last);
753  }
754  });
755 }
