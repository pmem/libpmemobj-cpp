1066 template <typename T>
1067 void
1068 vector<T>::_alloc(size_type capacity_new)
1069 {
1070  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
1071  assert(_data == nullptr);
1072  assert(_size == 0);
1073 
1074  if (capacity_new > max_size())
1075  throw std::length_error("New capacity exceeds max size.");
1076 
1077  _capacity = capacity_new;
1078 
1079  if (capacity_new == 0)
1080  return;

1082  /*
1083  * We need to cache pmemobj_tx_alloc return value and only after that
1084  * assign it to _data, because when pmemobj_tx_alloc fails, it aborts
1085  * transaction.
1086  */

1088  pmemobj_tx_alloc(sizeof(value_type) * capacity_new,
1089  detail::type_num<value_type>());
1090 
1091  if (res == nullptr)
1092  throw transaction_alloc_error(
1093  "Failed to allocate persistent memory object");
1094 
1095  _data = res;
1096 }
