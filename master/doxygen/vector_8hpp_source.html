504 template <typename T>
505 void
506 vector<T>::assign(size_type count, const_reference value)
507 {
508  pool_base pb = get_pool();
509 
510  transaction::run(pb, [&] {
511  if (count <= capacity()) {
512  /*
513  * Reallocation is not needed. First, replace old
514  * elements with new ones in range [0, size()).
515  * Depending on count, either call remaining old
516  * elements destructors, or append more new elements.
517  */
518  size_type size_old = _size;
519  snapshot_data(0, size_old);
520 
521  std::fill_n(
522  &_data[0],
523  (std::min)(count,
524  static_cast<size_type>(size_old)),
525  value);
526 
527  if (count > size_old) {
528 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
529  /*
530  * Range of memory:
531  * [&_data[size_old], &_data[count])
532  * is undefined, there is no need to snapshot
533  * and eventually rollback old data.
534  */
535  VALGRIND_PMC_ADD_TO_TX(
536  &_data[size_old],
537  sizeof(T) * (count - size_old));
538 #endif
539 
540  construct(size_old, count - size_old, value);
541  /*
542  * XXX: explicit persist is required here
543  * because given range wasn't snapshotted and
544  * won't be persisted automatically on tx
545  * commit. This can be changed once we will have
546  * implemented "uninitialized" flag for
547  * pmemobj_tx_xadd in libpmemobj.
548  */
549  pb.persist(&_data[size_old],
550  sizeof(T) * (count - size_old));
551  } else {
552  shrink(count);
553  }
554  } else {
555  dealloc();
556  alloc(count);
557  construct(0, count, value);
558  }
559  });
560 }

580 template <typename T>
581 template <typename InputIt,
582  typename std::enable_if<detail::is_input_iterator<InputIt>::value,
583  InputIt>::type *>
584 void
585 vector<T>::assign(InputIt first, InputIt last)
586 {
587  pool_base pb = get_pool();
588 
589  size_type size_new = static_cast<size_type>(std::distance(first, last));
590 
591  transaction::run(pb, [&] {
592  if (size_new <= capacity()) {
593  /*
594  * Reallocation is not needed. First, replace old
595  * elements with new ones in range [0, size()).
596  * Depending on size_new, either call remaining old
597  * elements destructors, or append more new elements.
598  */
599  size_type size_old = _size;
600  snapshot_data(0, size_old);
601 
602  InputIt mid = last;
603  bool growing = size_new > size_old;
604 
605  if (growing) {
606 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
607  /*
608  * Range of memory:
609  * [&_data[size_old], &_data[size_new])
610  * is undefined, there is no need to snapshot
611  * and eventually rollback old data.
612  */
613  VALGRIND_PMC_ADD_TO_TX(
614  &_data[size_old],
615  sizeof(T) * (size_new - size_old));
616 #endif
617 
618  mid = first;
619  std::advance(mid, size_old);
620  }
621 
622  iterator shrink_to = std::copy(first, mid, &_data[0]);
623 
624  if (growing) {
625  construct_range_copy(size_old, mid, last);
626  /*
627  * XXX: explicit persist is required here
628  * because given range wasn't snapshotted and
629  * won't be persisted automatically on tx
630  * commit. This can be changed once we will have
631  * implemented "uninitialized" flag for
632  * pmemobj_tx_xadd in libpmemobj.
633  */
634  pb.persist(&_data[size_old],
635  sizeof(T) * (size_new - size_old));
636  } else {
637  shrink(static_cast<size_type>(std::distance(
638  iterator(&_data[0]), shrink_to)));
639  }
640  } else {
641  dealloc();
642  alloc(size_new);
643  construct_range_copy(0, first, last);
644  }
645  });
646 }

1544 void
1545 vector<T>::insert_gap(size_type idx, size_type count)
1546 {
1547  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
1548 
1549  if (_capacity >= _size + count) {
1550  pointer dest = &_data[size() + count];
1551  pointer begin = &_data[idx];
1552  pointer end = &_data[size()];
1553 
1554  /*
1555  * XXX: There is no necessity to snapshot uninitialized data, so
1556  * we can optimize it by calling:
1557  * transaction::snapshot<T>(begin, size() - idx).
1558  * However, we need libpmemobj support for that, because right
1559  * now pmemcheck will report an error (uninitialized part of
1560  * data not added to tx).
1561  *
1562  * XXX: future optimization: we don't have to snapshot data
1563  * which we will not overwrite
1564  */
1565 #if LIBPMEMOBJ_CPP_VG_MEMCHECK_ENABLED
1566  VALGRIND_MAKE_MEM_DEFINED(end, sizeof(T) * count);
1567 #endif
1568  transaction::snapshot<T>(begin, size() + count);
1569 
1570  std::move_backward(begin, end, dest);
1571  } else {
1572  /*
1573  * XXX: future optimization: we don't have to snapshot data
1574  * which we will not overwrite
1575  */
1576  snapshot_data(0, _size);
1577 
1578  auto old_data = _data;
1579  auto old_size = _size;
1580  pointer old_begin = &_data[0];
1581  pointer old_mid = &_data[idx];
1582  pointer old_end = &_data[size()];
1583 
1584  _data = nullptr;
1585  _size = _capacity = 0;
1586 
1587  alloc(get_recommended_capacity(old_size + count));

1589  construct_range(0, old_begin, old_mid);
1590  construct_range(idx + count, old_mid, old_end);

1594  detail::destroy<value_type>(old_data[i]);
