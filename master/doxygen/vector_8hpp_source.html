654 template <typename T>
655 void
656 vector<T>::assign(size_type count, const_reference value)
657 {
658  pool_base pb = get_pool();
659 
660  transaction::run(pb, [&] {
661  if (count <= capacity()) {
662  /*
663  * Reallocation is not needed. First, replace old
664  * elements with new ones in range [0, size()).
665  * Depending on count, either call remaining old
666  * elements destructors, or append more new elements.
667  */
668  size_type size_old = _size;
669  snapshot_data(0, size_old);
670 
671  std::fill_n(
672  &_data[0],
673  (std::min)(count,
674  static_cast<size_type>(size_old)),
675  value);
676 
677  if (count > size_old) {
678 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
679  /*
680  * Range of memory:
681  * [&_data[size_old], &_data[count])
682  * is undefined, there is no need to snapshot
683  * and eventually rollback old data.
684  */
685  VALGRIND_PMC_ADD_TO_TX(
686  &_data[static_cast<difference_type>(
687  size_old)],
688  sizeof(T) * (count - size_old));
689 #endif
690 
691  construct_at_end(count - size_old, value);
692  /*
693  * XXX: explicit persist is required here
694  * because given range wasn't snapshotted and
695  * won't be persisted automatically on tx
696  * commit. This can be changed once we will have
697  * implemented "uninitialized" flag for
698  * pmemobj_tx_xadd in libpmemobj.
699  */
700  pb.persist(&_data[static_cast<difference_type>(
701  size_old)],
702  sizeof(T) * (count - size_old));
703  } else {
704  shrink(count);
705  }
706  } else {
707  dealloc();
708  alloc(count);
709  construct_at_end(count, value);
710  }
711  });
712 }

733 template <typename InputIt,
734  typename std::enable_if<detail::is_input_iterator<InputIt>::value,
735  InputIt>::type *>
736 void
737 vector<T>::assign(InputIt first, InputIt last)
738 {
739  pool_base pb = get_pool();
740 
741  size_type size_new = static_cast<size_type>(std::distance(first, last));
742 
743  transaction::run(pb, [&] {
744  if (size_new <= capacity()) {
745  /*
746  * Reallocation is not needed. First, replace old
747  * elements with new ones in range [0, size()).
748  * Depending on size_new, either call remaining old
749  * elements destructors, or append more new elements.
750  */
751  size_type size_old = _size;
752  snapshot_data(0, size_old);
753 
754  InputIt mid = last;
755  bool growing = size_new > size_old;
756 
757  if (growing) {
758 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
759  /*
760  * Range of memory:
761  * [&_data[size_old], &_data[size_new])

764  */
765  VALGRIND_PMC_ADD_TO_TX(
766  &_data[static_cast<difference_type>(
767  size_old)],
768  sizeof(T) * (size_new - size_old));
769 #endif
770 
771  mid = first;
772  std::advance(mid, size_old);
773  }
774 
775  iterator shrink_to = std::copy(first, mid, &_data[0]);
776 
777  if (growing) {
778  construct_at_end(mid, last);
779  /*
780  * XXX: explicit persist is required here
781  * because given range wasn't snapshotted and
782  * won't be persisted automatically on tx

784  * implemented "uninitialized" flag for
785  * pmemobj_tx_xadd in libpmemobj.
786  */
