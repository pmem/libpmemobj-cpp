583  {
584  /*
585  * We cannot do anything when in TX_STAGE_NONE because

589  if (obj_stage == TX_STAGE_NONE)

591

593  if (data == nullptr)
594  return;
595

598

603  if (obj_stage == TX_STAGE_FINALLY) {
604  delete data;
