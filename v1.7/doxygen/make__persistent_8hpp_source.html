77 template <typename T, typename... Args>
78 typename detail::pp_if_not_array<T>::type
79 make_persistent(allocation_flag flag, Args &&... args)
80 {
81  if (pmemobj_tx_stage() != TX_STAGE_WORK)
82  throw transaction_scope_error(
83  "refusing to allocate memory outside of transaction scope");
84 
85  persistent_ptr<T> ptr =
86  pmemobj_tx_xalloc(sizeof(T), detail::type_num<T>(), flag.value);
87 
88  if (ptr == nullptr)
89  throw transaction_alloc_error(
90  "failed to allocate persistent memory object");
91 
92  detail::create<T, Args...>(ptr.get(), std::forward<Args>(args)...);
93 
94  return ptr;
95 }

136 template <typename T>
137 void
138 delete_persistent(typename detail::pp_if_not_array<T>::type ptr)
139 {
140  if (pmemobj_tx_stage() != TX_STAGE_WORK)
141  throw transaction_scope_error(
142  "refusing to free memory outside of transaction scope");
143 
144  if (ptr == nullptr)
145  return;
146 
147  /*
148  * At this point, everything in the object should be tracked
149  * and reverted on transaction abort.
150  */
151  detail::destroy<T>(*ptr);
152 
153  if (pmemobj_tx_free(*ptr.raw_ptr()) != 0)
154  throw transaction_free_error(
155  "failed to delete persistent memory object");
156 }
